Deployment Manifest


apiVersion: apps/v1
kind: Deployment
metadata:
  name: k8s-demo-vx
  labels:
    app: k8s-demo-vx
    app.kubernetes.io/name: k8s-demo-vx
    app.kubernetes.io/component: server
spec:
  progressDeadlineSeconds: 600
  replicas: 1
  revisionHistoryLimit: 10
  selector:
    matchLabels:
      app: k8s-demo-vx
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxSurge: 25%
      maxUnavailable: 25%
  template:
    metadata:
      labels:
        app: k8s-demo-vx
        app.kubernetes.io/name: k8s-demo-vx
        app.kubernetes.io/component: server
    spec:
      containers:
        - name: backend
          image: ghcr.io/stefan-niemeyer/k8s-demo/k8s-demo:v1
          imagePullPolicy: IfNotPresent
          ports:
            - name: http-app
              containerPort: 3000
              protocol: TCP
          volumeMounts:
            - name: npm-logs
              mountPath: /home/node/.npm
          livenessProbe:
            failureThreshold: 3
            httpGet:
              path: /health
              port: http-app
              scheme: HTTP
            initialDelaySeconds: 10
            periodSeconds: 10
            successThreshold: 1
            timeoutSeconds: 3
          readinessProbe:
            failureThreshold: 1
            httpGet:
              path: /ready
              port: http-app
              scheme: HTTP
            periodSeconds: 10
            initialDelaySeconds: 5
            successThreshold: 1
            timeoutSeconds: 3
          resources:
            requests:
              cpu: 7m
              memory: 200Mi
            limits:
              cpu: 100m
              memory: 1024Mi
          securityContext:
            readOnlyRootFilesystem: true
            allowPrivilegeEscalation: false
            privileged: false
            runAsNonRoot: true
      enableServiceLinks: false
      automountServiceAccountToken: false
      volumes:
        - name: npm-logs
          emptyDir: {}

Showtime Deployment


            # Deployment anlegen
            $ k apply -f k8s-demo-vx-deploy.yaml
            deployment.apps/k8s-demo-vx created
        

            # Deployment anzeigen
            $ k get deployments -w
            NAME          READY   UP-TO-DATE   AVAILABLE   AGE
            k8s-demo-vx   0/1     1            0           17s
            k8s-demo-vx   1/1     1            1           21s
            # Mit Strg-C beenden

Arbeiten mit dem Pod

Bonusaufgaben Deployment


            # Name des Pods mittels Umgebungsvariable POD_NAME merken
            $ setpod
Parameter -l oder --selector erläutern

            # Befehl im Pod ausführen
            $ k exec $POD_NAME -- env
        

            # Shell im Pod öffnen
            $ k exec -it $POD_NAME -- bash
        

            # Im Pod (Der Prompt sieht anders aus)
            $ ps
            $ cat src/server.js
            $ curl -w " -> %{response_code}\n" http://localhost:3000/state
        

            # Shell im Pod beenden
            $ exit
